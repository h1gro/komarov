# Краткая инструкция по созданию SSH-ключа

```bash
# 1. Генерация ключа (выберите один вариант)
ssh-keygen -t ed25519 -C "your_email@example.com"  # Рекомендуемый современный алгоритм

# 2. Нажмите Enter для сохранения в стандартный путь (~/.ssh/id_algorithm)
# 3. Введите пароль (или оставьте пустым)
# 4. Добавьте ключ в ssh-agent:
eval "$(ssh-agent -s)"
ssh-add ~/.ssh/id_ed25519

# 5. Скопируйте публичный ключ в буфер обмена:
cat ~/.ssh/id_ed25519.pub | xclip -sel clip
